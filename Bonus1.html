class Worker:
    def __init__(self, x, y, q):
        self.x = x # s-by-d local feature matrix
        self.y = y # s-by-1 local label matrix
        self.q = q
        self.s = x.shape[0] # number of local samples
        self.d = x.shape[1] # number of features
        self.w = numpy.zeros((d,1)) # d-by-1 model parameter vector
        
    # Set the model parameters to the latest
    def set_param(self, w):
        self.w = w
    
    def objective(self, lam):
        yx = numpy.multiply(self.y, self.x) # s-by-d matrix
        yxw = numpy.dot(yx, self.w) # s-by-1 matrix
        vec1 = numpy.exp(yxw) # s-by-1 matrix
        vec2 = numpy.divide(yx, 1+vec1) # s-by-d matrix
        loss = numpy.mean(vec2)
        reg = lam / 2* numpy.sum(self.w * self.w)
        return loss+reg
    
    def gd_gradient(self, lam):
        yx = numpy.multiply(self.y, self.x) # s-by-d matrix
        yxw = numpy.dot(yx, self.w) # s-by-1 matrix
        vec1 = numpy.exp(yxw) # s-by-1 matrix
        vec2 = numpy.divide(yx, 1+vec1) # s-by-d matrix
        vec3 = -numpy.mean(vec2, axis=0).reshape(self.d, 1) # d-by-1 matrix
        g = vec3+ lam * self.w
        return g

    def sgd_gradient(self, lam):
        yx = self.y * self.x
        yxw = float(numpy.dot(yx, self.w))
        #loss = numpy.log(1+numpy.exp(-yxw))
        #reg = -lam / 2 * numpy.sum(self.w * self.w)
        #obj = loss + reg
    
        g_loss = -yx.T / (1 + numpy.exp(yxw))
        g = g_loss + lam * self.w
    
        return g

    def loss(self):
        yx = numpy.multiply(self.y, self.x) # s-by-d matrix
        yxw = numpy.dot(yx, self.w) # s-by-1 matrix
        vec1 = numpy.exp(-yxw) # s-by-1 matrix
        vec2 = numpy.log(1+vec1)
        return numpy.sum(vec2)
        
        
    # Compute the local gradient(GD)
    def GD(self, lam):
        stepsize = 1
        #objvals = numpy.zeros(self.q) # store the objective value
        
        for r in range(self.q):
            #objval = objective(lam)
            #objvals[r] = objval
            gd = self.gd_gradient(lam)
            self.w -= stepsize * gd
        
        return self.w
    
        
    # Compute the local gradient(SGD)
    def SGD(self, lam):
        stepsize = 0.1
        #objvals = numpy.zeros(self.q)
        for t in range(self.q):
            # randomly shuffle the samples
            rand_indices = numpy.random.permutation(self.s)
            x_rand = self.x[rand_indices, :]

1.6 Server Class for FedAvg¶

class Server:
    def __init__(self, m, n, d, q):
        self.m = m # number of work nodes
        self.q = q
        self.n = n # number of training samples
        self.d = d # number of features
        self.w = numpy.zeros((d,1)) # d-by-1 model parameter vector\
        self.g = numpy.zeros((d,1)) # d-by-1 gradient  
        self.loss = 0 # loss function value
        self.obj = 0 # objective function value
    
    def broadcast(self):
        return self.w
    
    # Sum the gradients and loss functions evaluated by the workers
    # Args:
    #   grads: a list of d-by-1 vectors
    #   losses: a list of scalars
    def aggregate(self, grads, losses):
        self.g = numpy.zeros((self.d, 1))
        self.loss = 0
        for k in range(self.m):
            self.g += grads[k]
            self.loss += losses[k]
        
    # Compute the gradient (from the loss and regularization)
    def gradient(self):
        self.g = self.g / self.m
        
    # objective function
    def objective(self, lam):
        reg = lam / 2* numpy.sum(self.w * self.w)
        self.obj = self.loss / self.n + reg
        return self.obj

    
    # Update the model parameter using GD/SGD
    # Args:
    #   alpha: learning rate
    #   : a list of scalars
    def update(self, alpha):
        self.w -= alpha * self.g

import math

# Create a server and m worker nodes
def create_server_workers(m,x,y,q):
    n,d = x.shape
    s = math.floor(n/m)
    server = Server(m,n,d,q)
    workers = []
    
    for i in range(m):
        indices = list(range(i*s, (i+1)*s))
        worker = Worker(x[indices,:],y[indices,:], q)
        workers.append(worker)
    
    return server, workers
